- [ ] **Step 1: Write MovingPlatform.cs**

```csharp
using UnityEngine;

/// <summary>
/// Kinematic platform ping-ponging between two points at constant speed.
/// Velocity-driven (not MovePosition) so PlayerMotor can read the platform's
/// velocity off attachedRigidbody and carry a player standing on it.
/// </summary>
[RequireComponent(typeof(Rigidbody2D))]
public class MovingPlatform : MonoBehaviour
{
    [SerializeField] private Vector2 pointA;
    [SerializeField] private Vector2 pointB;
    [SerializeField] private float speed = 1.5f;

    private Rigidbody2D body;
    private Vector2 target;

    private void Awake()
    {
        body = GetComponent<Rigidbody2D>();
        target = pointB;
    }

    private void FixedUpdate()
    {
        Vector2 toTarget = target - body.position;
        float step = speed * Time.fixedDeltaTime;
        if (toTarget.magnitude <= step)
        {
            // Land exactly on the endpoint, pause one physics step, turn around.
            body.linearVelocity = Vector2.zero;
            body.position = target;
            target = target == pointA ? pointB : pointA;
            return;
        }
        body.linearVelocity = toTarget.normalized * speed;
    }

    private void OnDrawGizmosSelected()
    {
        Gizmos.color = Color.cyan;
        Gizmos.DrawLine(pointA, pointB);
    }
}
```

```csharp
    private void FixedUpdate()
    {
        Vector2 checkCenter = (Vector2)transform.position + Vector2.down * groundCheckOffset;
        Collider2D ground = Physics2D.OverlapBox(checkCenter, groundCheckSize, 0f, groundLayer);
        IsGrounded = ground != null;

        // Standing on a moving platform: the direct velocity write below would
        // let the platform slide out from under us, so the ground's own
        // velocity is added to the walk velocity while grounded. Static ground
        // has no attachedRigidbody -> contributes 0.
        groundVelocityX = 0f;
        if (ground != null && ground.attachedRigidbody != null)
            groundVelocityX = ground.attachedRigidbody.linearVelocity.x;

        if (IsDashing)
        {
            // Move input is intentionally ignored while dashing (spec).
            body.linearVelocity = new Vector2(dashSpeed * dashDirection, 0f);
            dashTimeLeft -= Time.fixedDeltaTime;
            if (!IsDashing) body.gravityScale = defaultGravityScale;
        }
        else
        {
            body.linearVelocity = new Vector2(
                moveInput * moveSpeed + groundVelocityX, body.linearVelocity.y);
        }
    }
```

```csharp
using UnityEditor;
using UnityEngine;
using UnityEngine.InputSystem;
using UnityEngine.Rendering.Universal;

/// <summary>
/// Shared building blocks for the scene-builder menu items. Each concrete
/// builder (TestSceneBuilder, Level1SceneBuilder) composes these into a
/// scene; only scene-specific geometry and wiring live in the builders.
/// </summary>
internal static class SceneBuilderCommon
{
    public const string GroundLayerName = "Ground";
    public const string ActionsPath = "Assets/InputSystem_Actions.inputactions";
    // Unity's built-in 1x1-unit square (the GameObject > 2D Object > Sprites > Square asset).
    private const string SquareSpritePath =
        "Packages/com.unity.2d.sprite/Editor/ObjectMenuCreation/DefaultAssets/Textures/v2/Square.png";
    // TMP's default font resources ship inside the uGUI package but must be
    // imported into Assets once per project.
    private const string TmpEssentialsPackagePath =
        "Packages/com.unity.ugui/Package Resources/TMP Essential Resources.unitypackage";
    private const string TmpSettingsAssetPath = "Assets/TextMesh Pro/Resources/TMP Settings.asset";

    public static readonly Color GroundColor = new Color(0.35f, 0.35f, 0.35f);
    public static readonly Color PlatformColor = new Color(0.55f, 0.55f, 0.55f);
    public static readonly Color MovingPlatformColor = new Color(0.45f, 0.65f, 0.9f);
    public static readonly Color GoalColor = new Color(0.25f, 0.85f, 0.35f);

    public static void EnsureTmpEssentials()
    {
        // Runtime-created TextMeshPro components read the default font from TMP
        // Settings; without the import TMP logs "Can't Generate Mesh, No Font
        // Asset has been assigned" and world/overlay text renders nothing.
        if (AssetDatabase.LoadMainAssetAtPath(TmpSettingsAssetPath) != null)
            return;
        // Note: ImportPackage is asynchronous — fine today because scenes bake
        // no TMP objects (all text UI builds itself at runtime).
        AssetDatabase.ImportPackage(
            System.IO.Path.GetFullPath(TmpEssentialsPackagePath), false);
        Debug.Log("Imported TMP Essential Resources (required for text UI).");
    }

    public static void EnsureGroundLayer()
    {
        var tagManager = new SerializedObject(
            AssetDatabase.LoadAllAssetsAtPath("ProjectSettings/TagManager.asset")[0]);
        SerializedProperty layers = tagManager.FindProperty("layers");

        for (int i = 8; i < layers.arraySize; i++)
            if (layers.GetArrayElementAtIndex(i).stringValue == GroundLayerName)
                return;

        for (int i = 8; i < layers.arraySize; i++)
        {
            SerializedProperty slot = layers.GetArrayElementAtIndex(i);
            if (string.IsNullOrEmpty(slot.stringValue))
            {
                slot.stringValue = GroundLayerName;
                tagManager.ApplyModifiedProperties();
                return;
            }
        }
        throw new System.InvalidOperationException("No free layer slot for the Ground layer.");
    }

    public static Sprite LoadSquareSprite()
    {
        var sprite = AssetDatabase.LoadAssetAtPath<Sprite>(SquareSpritePath);
        if (sprite == null)
            throw new System.InvalidOperationException(
                $"Built-in square sprite not found at {SquareSpritePath} — did the com.unity.2d.sprite package layout change?");
        return sprite;
    }

    public static void EnsureSceneInBuildSettings(string scenePath)
    {
        // A player build only contains scenes listed here.
        var scenes = new System.Collections.Generic.List<EditorBuildSettingsScene>(
            EditorBuildSettings.scenes);
        if (scenes.Exists(s => s.path == scenePath))
            return;
        scenes.Add(new EditorBuildSettingsScene(scenePath, true));
        EditorBuildSettings.scenes = scenes.ToArray();
    }

    public static GameObject BuildCamera(Vector3 position, float orthographicSize)
    {
        var go = new GameObject("Main Camera") { tag = "MainCamera" };
        var cam = go.AddComponent<Camera>();
        cam.orthographic = true;
        cam.orthographicSize = orthographicSize;
        cam.clearFlags = CameraClearFlags.SolidColor;
        cam.backgroundColor = new Color(0.12f, 0.12f, 0.15f);
        go.transform.position = position;
        go.AddComponent<AudioListener>();
        return go;
    }

    public static void BuildGlobalLight()
    {
        // URP 2D's default sprite material is lit; without a global light
        // everything renders black.
        var go = new GameObject("Global Light 2D");
        var light = go.AddComponent<Light2D>();
        light.lightType = Light2D.LightType.Global;
        light.intensity = 1f;
    }

    public static void BuildBox(string name, Sprite sprite, Vector2 position,
        Vector2 size, int layer, Color color)
    {
        var go = new GameObject(name) { layer = layer, isStatic = true };
        go.transform.position = position;
        go.transform.localScale = new Vector3(size.x, size.y, 1f);
        var renderer = go.AddComponent<SpriteRenderer>();
        renderer.sprite = sprite;
        renderer.color = color;
        go.AddComponent<BoxCollider2D>(); // auto-sizes to the 1x1 sprite, scaled by transform
    }

    public static GameObject BuildTrigger(string name, Vector2 position, Vector2 size)
    {
        var go = new GameObject(name);
        go.transform.position = position;
        var box = go.AddComponent<BoxCollider2D>();
        box.isTrigger = true;
        box.size = size;
        return go;
    }

    public static void Wire(Component component, string field, Object value)
    {
        var so = new SerializedObject(component);
        so.FindProperty(field).objectReferenceValue = value;
        so.ApplyModifiedPropertiesWithoutUndo();
    }

    public static GameObject BuildPlayer(Sprite sprite, Vector2 position, int groundLayer)
    {
        var go = new GameObject("Player");
        go.transform.position = position;

        var renderer = go.AddComponent<SpriteRenderer>();
        renderer.sprite = sprite;
        renderer.sortingOrder = 5;

        var body = go.AddComponent<Rigidbody2D>();
        body.gravityScale = 4f; // motor tuning (jumpVelocity 14) assumes this
        body.freezeRotation = true;
        body.interpolation = RigidbodyInterpolation2D.Interpolate;
        body.collisionDetectionMode = CollisionDetectionMode2D.Continuous;

        go.AddComponent<BoxCollider2D>();

        var motor = go.AddComponent<PlayerMotor>();
        var motorSo = new SerializedObject(motor);
        motorSo.FindProperty("groundLayer").intValue = 1 << groundLayer;
        motorSo.ApplyModifiedPropertiesWithoutUndo();

        var queue = go.AddComponent<CommandQueue>();
        var queueSo = new SerializedObject(queue);
        queueSo.FindProperty("motor").objectReferenceValue = motor;
        queueSo.ApplyModifiedPropertiesWithoutUndo();

        var input = go.AddComponent<PlayerInputHandler>();
        var inputSo = new SerializedObject(input);
        inputSo.FindProperty("actions").objectReferenceValue =
            AssetDatabase.LoadAssetAtPath<InputActionAsset>(ActionsPath);
        inputSo.ApplyModifiedPropertiesWithoutUndo();

        var ui = go.AddComponent<CommandQueueUI>();
        var uiSo = new SerializedObject(ui);
        uiSo.FindProperty("queue").objectReferenceValue = queue;
        uiSo.ApplyModifiedPropertiesWithoutUndo();

        return go;
    }

    public static CameraFollow WireCameraFollow(GameObject player,
        Vector2 boundsMin, Vector2 boundsMax)
    {
        var camera = GameObject.FindWithTag("MainCamera");
        var follow = camera.AddComponent<CameraFollow>();
        var followSo = new SerializedObject(follow);
        followSo.FindProperty("target").objectReferenceValue = player.transform;
        followSo.FindProperty("boundsMin").vector2Value = boundsMin;
        followSo.FindProperty("boundsMax").vector2Value = boundsMax;
        followSo.ApplyModifiedPropertiesWithoutUndo();
        // Start the camera near its follow position (CameraFollow.Start snaps
        // to the clamped position anyway).
        camera.transform.position =
            player.transform.position + followSo.FindProperty("offset").vector3Value;
        return follow;
    }

    public static LevelManager BuildLevelManager(GameObject player, CameraFollow follow)
    {
        var managerGo = new GameObject("LevelManager");
        var banner = managerGo.AddComponent<BannerUI>();
        var manager = managerGo.AddComponent<LevelManager>();

        var managerSo = new SerializedObject(manager);
        managerSo.FindProperty("motor").objectReferenceValue =
            player.GetComponent<PlayerMotor>();
        managerSo.FindProperty("queue").objectReferenceValue =
            player.GetComponent<CommandQueue>();
        managerSo.FindProperty("inputHandler").objectReferenceValue =
            player.GetComponent<PlayerInputHandler>();
        managerSo.FindProperty("cameraFollow").objectReferenceValue = follow;
        managerSo.FindProperty("banner").objectReferenceValue = banner;
        managerSo.ApplyModifiedPropertiesWithoutUndo();
        return manager;
    }

    public static Checkpoint BuildCheckpoint(string name, Vector2 position, LevelManager manager)
    {
        var go = BuildTrigger(name, position, new Vector2(1f, 3f));
        var checkpoint = go.AddComponent<Checkpoint>();
        Wire(checkpoint, "levelManager", manager);
        return checkpoint;
    }

    public static void SetInitialCheckpoint(LevelManager manager, Checkpoint checkpoint)
    {
        Wire(manager, "initialCheckpoint", checkpoint);
    }

    public static void BuildKillZone(Vector2 center, Vector2 size, LevelManager manager)
    {
        var go = BuildTrigger("Kill Zone", center, size);
        Wire(go.AddComponent<KillZone>(), "levelManager", manager);
    }

    public static void BuildGoal(Sprite square, Vector2 position, Vector2 scale, LevelManager manager)
    {
        // Visible green block; collider auto-sizes to the 1x1 sprite, scaled.
        var go = new GameObject("Goal");
        go.transform.position = position;
        go.transform.localScale = new Vector3(scale.x, scale.y, 1f);
        var renderer = go.AddComponent<SpriteRenderer>();
        renderer.sprite = square;
        renderer.color = GoalColor;
        go.AddComponent<BoxCollider2D>().isTrigger = true;
        Wire(go.AddComponent<LevelGoal>(), "levelManager", manager);
    }

    public static void BuildMovingPlatform(string name, Sprite sprite,
        Vector2 pointA, Vector2 pointB, Vector2 size, float speed, int layer)
    {
        var go = new GameObject(name) { layer = layer };
        go.transform.position = pointA;
        go.transform.localScale = new Vector3(size.x, size.y, 1f);
        var renderer = go.AddComponent<SpriteRenderer>();
        renderer.sprite = sprite;
        renderer.color = MovingPlatformColor;
        go.AddComponent<BoxCollider2D>();

        var body = go.AddComponent<Rigidbody2D>();
        body.bodyType = RigidbodyType2D.Kinematic;
        body.interpolation = RigidbodyInterpolation2D.Interpolate;

        var platform = go.AddComponent<MovingPlatform>();
        var so = new SerializedObject(platform);
        so.FindProperty("pointA").vector2Value = pointA;
        so.FindProperty("pointB").vector2Value = pointB;
        so.FindProperty("speed").floatValue = speed;
        so.ApplyModifiedPropertiesWithoutUndo();
    }
}
```

```csharp
/// <summary>
/// Builds the graybox test scene from code so it is reproducible and
/// reviewable. Run via Tools > Input Lag > Build Test Scene. Re-running
/// overwrites the scene. Scene-specific geometry only — shared building
/// blocks live in SceneBuilderCommon.

    public static void Build()
    {
        // Don't silently discard someone's unsaved scene edits.
        if (!EditorSceneManager.SaveCurrentModifiedScenesIfUserWantsTo())
            return;

        SceneBuilderCommon.EnsureTmpEssentials();
        SceneBuilderCommon.EnsureGroundLayer();
        Sprite square = SceneBuilderCommon.LoadSquareSprite();
        int groundLayer = LayerMask.NameToLayer(SceneBuilderCommon.GroundLayerName);

        SceneBuilderCommon.BuildBox("Ground Right", square, new Vector2(11.5f, -0.5f),
            new Vector2(5f, 1f), groundLayer, SceneBuilderCommon.GroundColor);

        // inside the ~2.5-unit jump height.
```

```csharp
using UnityEditor;
using UnityEditor.SceneManagement;
using UnityEngine;

/// <summary>
/// Builds Level 1 "Think Ahead" from the level design spec
/// (docs/superpowers/specs/2026-07-24-level-1-design.md). Eight teaching
/// rooms, delay 2 s, no deaths (recovery pits everywhere). Every number
/// below comes from that spec — retune the motor and these must be
/// re-derived. Run via Tools > Input Lag > Build Level 1.
/// </summary>
public static class Level1SceneBuilder
{
    private const string ScenePath = "Assets/Scenes/Level1.unity";

    private static Sprite square;
    private static int groundLayer;

    [MenuItem("Tools/Input Lag/Build Level 1")]
    public static void Build()
    {
        if (!EditorSceneManager.SaveCurrentModifiedScenesIfUserWantsTo())
            return;

        SceneBuilderCommon.EnsureTmpEssentials();
        SceneBuilderCommon.EnsureGroundLayer();
        square = SceneBuilderCommon.LoadSquareSprite();
        groundLayer = LayerMask.NameToLayer(SceneBuilderCommon.GroundLayerName);

        var scene = EditorSceneManager.NewScene(NewSceneSetup.EmptyScene, NewSceneMode.Single);

        SceneBuilderCommon.BuildCamera(new Vector3(1f, 3f, -10f), 5.5f);
        SceneBuilderCommon.BuildGlobalLight();

        BuildGeometry();
        BuildFerries();

        GameObject player = SceneBuilderCommon.BuildPlayer(square, new Vector2(1f, 1f), groundLayer);
        // Level rule: delay = 2 s. Set explicitly so the intent survives any
        // future change to the CommandQueue default.
        var queueSo = new SerializedObject(player.GetComponent<CommandQueue>());
        queueSo.FindProperty("delaySeconds").floatValue = 2f;
        queueSo.ApplyModifiedPropertiesWithoutUndo();

        CameraFollow follow = SceneBuilderCommon.WireCameraFollow(player,
            new Vector2(-1f, -4f), new Vector2(163f, 20f));

        LevelManager manager = SceneBuilderCommon.BuildLevelManager(player, follow);
        BuildCheckpoints(manager);
        // Safety net only — every pit has a floor; this should be unreachable.
        SceneBuilderCommon.BuildKillZone(new Vector2(80f, -12f), new Vector2(170f, 2f), manager);
        SceneBuilderCommon.BuildGoal(square, new Vector2(158f, 10.95f), new Vector2(1f, 1.5f), manager);

        EditorSceneManager.SaveScene(scene, ScenePath);
        SceneBuilderCommon.EnsureSceneInBuildSettings(ScenePath);
        Debug.Log($"Level 1 built and saved to {ScenePath}");
    }

    private static void BuildGeometry()
    {
        Color G = SceneBuilderCommon.GroundColor;
        Color S = SceneBuilderCommon.PlatformColor; // pit escape steps, lighter

        // Boundaries
        Floor("Wall Left", -2f, 0f, 6f, -4f, G);
        Floor("Wall Right", 162f, 164f, 18f, 6.2f, G);

        // Room 1 - The Late Jump: flat runway, then a chest-high block.
        Floor("R1 Floor", 0f, 14f, 0f, -4f, G);
        Floor("R1 Block / R2 Floor", 14f, 22f, 2f, -2f, G);

        // Room 2 - The Planned Gap (3.0): gap x22-25, recovery pit below.
        Floor("R2 Pit Floor", 22f, 25f, -1f, -2f, G);
        Floor("R2 Pit Step", 22f, 23.5f, 0.5f, -1f, S);
        Floor("R2 Far Floor", 25f, 34f, 2f, -2f, G);

        // Room 3 - Stairs of Intent: two 2.0 rises.
        Floor("R3 Step A", 34f, 39f, 4f, -2f, G);
        Floor("R3 Step B", 39f, 45f, 6f, -2f, G);

        // Room 4 - The Ferry: gap x45-53 (8.0), ferry crosses it.
        Floor("R4 Pit Floor", 45f, 53f, 3f, 2f, G);
        Floor("R4 Pit Step", 45f, 46.5f, 4.5f, 3f, S);
        Floor("R4 Far Floor", 53f, 60f, 6f, 2f, G);

        // Room 5 - The Long Gap (7.0): jump falls short, jump->dash clears.
        Floor("R5 Pit Floor", 60f, 67f, 3f, 2f, G);
        Floor("R5 Pit Step", 60f, 61.5f, 4.5f, 3f, S);
        Floor("R5 Far Floor", 67f, 79f, 6f, 2f, G);

        // Room 6 - The Tower: 5.0 gap onto a ledge 2.2 up. Plain jump is at
        // ~1.1 height when it reaches the wall (needs 2.2) -> always bonks;
        // jump->dash carries 2.5 height across.
        Floor("R6 Pit Floor", 79f, 84f, 3f, 2f, G);
        Floor("R6 Pit Step", 79f, 80.5f, 4.5f, 3f, S);
        Floor("R6 Ledge", 84f, 91f, 8.2f, 2f, G);

        // Room 7 - Ferry + Combo: 15.0 gap; hop on ferry, combo off its far
        // extreme (6.25 to the lip: > jump 5.7, < combo 8.7).
        Floor("R7 Pit Floor", 91f, 106f, 5.2f, 4.2f, G);
        Floor("R7 Pit Step", 91f, 92.5f, 6.7f, 5.2f, S);
        Floor("R7 Far / R8 Entry", 106f, 116f, 8.2f, 4.2f, G);

        // Room 8 - Final Exam: step up, easy gap, ferry, combo gap, goal.
        Floor("R8 Step Block", 116f, 124f, 10.2f, 4.2f, G);
        Floor("R8b Pit Floor", 124f, 127.5f, 7.2f, 6.2f, G);
        Floor("R8b Pit Step", 124f, 125.5f, 8.7f, 7.2f, S);
        Floor("R8b Floor", 127.5f, 134f, 10.2f, 6.2f, G);
        Floor("R8c Pit Floor", 134f, 142f, 7.2f, 6.2f, G);
        Floor("R8c Pit Step", 134f, 135.5f, 8.7f, 7.2f, S);
        Floor("R8d Floor", 142f, 148f, 10.2f, 6.2f, G);
        Floor("R8d Pit Floor", 148f, 155f, 7.2f, 6.2f, G);
        Floor("R8d Pit Step", 148f, 149.5f, 8.7f, 7.2f, S);
        Floor("Goal Floor", 155f, 162f, 10.2f, 6.2f, G);
    }

    private static void BuildFerries()
    {
        Vector2 ferrySize = new Vector2(3.5f, 0.5f);
        // Every leg = travel / speed = exactly 2.0 s — the delay IS the
        // prediction rule: "press when it's far; it arrives as the press fires."
        SceneBuilderCommon.BuildMovingPlatform("R4 Ferry", square,
            new Vector2(47.5f, 5.75f), new Vector2(50.5f, 5.75f), ferrySize, 1.5f, groundLayer);
        SceneBuilderCommon.BuildMovingPlatform("R7 Ferry", square,
            new Vector2(95f, 7.95f), new Vector2(98f, 7.95f), ferrySize, 1.5f, groundLayer);
        SceneBuilderCommon.BuildMovingPlatform("R8 Ferry", square,
            new Vector2(136.75f, 9.95f), new Vector2(139.25f, 9.95f), ferrySize, 1.25f, groundLayer);
    }

    private static void BuildCheckpoints(LevelManager manager)
    {
        Checkpoint first = SceneBuilderCommon.BuildCheckpoint("CP R1 (spawn)", new Vector2(1f, 1.5f), manager);
        SceneBuilderCommon.BuildCheckpoint("CP R2", new Vector2(16f, 3.5f), manager);
        SceneBuilderCommon.BuildCheckpoint("CP R3", new Vector2(26f, 3.5f), manager);
        SceneBuilderCommon.BuildCheckpoint("CP R4", new Vector2(40f, 7.5f), manager);
        SceneBuilderCommon.BuildCheckpoint("CP R5", new Vector2(54f, 7.5f), manager);
        SceneBuilderCommon.BuildCheckpoint("CP R6", new Vector2(68f, 7.5f), manager);
        SceneBuilderCommon.BuildCheckpoint("CP R7", new Vector2(85f, 9.7f), manager);
        SceneBuilderCommon.BuildCheckpoint("CP R8", new Vector2(107f, 9.7f), manager);
        SceneBuilderCommon.BuildCheckpoint("CP R8 mid", new Vector2(143f, 11.7f), manager);
        SceneBuilderCommon.SetInitialCheckpoint(manager, first);
    }

    /// <summary>Box from an (x0->x1, yTop->yBottom) slab — the spec's native format.</summary>
    private static void Floor(string name, float x0, float x1, float yTop, float yBottom, Color color)
    {
        var center = new Vector2((x0 + x1) * 0.5f, (yTop + yBottom) * 0.5f);
        var size = new Vector2(x1 - x0, yTop - yBottom);
        SceneBuilderCommon.BuildBox(name, square, center, size, groundLayer, color);
    }
}
```
